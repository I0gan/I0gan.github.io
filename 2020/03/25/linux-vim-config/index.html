vim config

"Author: I0gan

colorscheme koehler "主题
set cursorline "设置横线

syntax on "语法高亮
set autowrite "自动保存
set ruler     "设置尺子
set noeb      "去掉输入错误提示音
set tabstop=4 "设置缩进为4

"自动缩进
set autoindent

"set cindent
"统一缩进为4
set ts=4
set softtabstop=4
set shiftwidth=4


set number "显示行号
"set ignorecase "搜索时忽略大小写

"搜索时逐字高亮
set hlsearch  
set incsearch 
set enc=utf-8 "设置编码

"设置鼠标可以在任何地方都使用
set mouse=a 
set selection=exclusive
set selectmode=mouse,key


set completeopt=preview,menu "代码补全
set clipboard=unnamed        "共享剪贴板  
set autoread                 "设置当文件被改动时自动载入
set confirm                  "在处理未保存或只读文件的时候，弹出确认
cmap w!! w !sudo tee > /dev/null %


"退出再次打开,光标回到原来的位置
if has("autocmd")
           au BufReadPost * if line("'\"") > 1 && line("'\"") <= line("$") | exe "normal! g'\"" | endif
endif
